Exemplo de código em Python

numeros = []
min = max = 0
for pos in range(0,5):
    numeros.append(int(input(f'Digite um valor para a posição {pos + 1}: ')))
    if pos == 0:
        max = min = numeros[pos]
    else:
        if numeros[pos] > max:
            max = numeros[pos]
        elif numeros[pos] < min:
            min = numeros[pos]
